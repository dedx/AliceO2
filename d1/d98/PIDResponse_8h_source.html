102  void process(aod::Collision const& collision, soa::Join<aod::Tracks, aod::TracksExtra> const& tracks)
103  {
104  LOGF(info, "Tracks for collision: %d", tracks.size());
105  tof::EventTime evt = tof::EventTime();
106  evt.SetEvTime(0, collision.collisionTime());
107  evt.SetEvTimeReso(0, collision.collisionTimeRes());
108  evt.SetEvTimeMask(0, collision.collisionTimeMask());
109  tof::Response resp = tof::Response();
110  resp.SetEventTime(evt);
111  for (auto i : tracks) {
112  resp.UpdateTrack(i.p(), i.tofExpMom() / tof::Response::kCSPEED, i.length(), i.tofSignal());
113  tofpidbeta(resp.GetBeta(),
114  resp.GetBetaExpectedSigma(),
115  resp.GetExpectedBeta(PID::Electron),
116  resp.GetBetaExpectedSigma(),
117  resp.GetBetaNumberOfSigmas(PID::Electron));
118  tofpid(
119  resp.GetExpectedSignal(PID::Electron),
120  resp.GetExpectedSignal(PID::Muon),
121  resp.GetExpectedSignal(PID::Pion),
122  resp.GetExpectedSignal(PID::Kaon),
123  resp.GetExpectedSignal(PID::Proton),
124  resp.GetExpectedSignal(PID::Deuteron),
125  resp.GetExpectedSignal(PID::Triton),
126  resp.GetExpectedSignal(PID::Helium3),
127  resp.GetExpectedSignal(PID::Alpha),
128  resp.GetExpectedSigma(PID::Electron),
129  resp.GetExpectedSigma(PID::Muon),
130  resp.GetExpectedSigma(PID::Pion),
131  resp.GetExpectedSigma(PID::Kaon),
132  resp.GetExpectedSigma(PID::Proton),
133  resp.GetExpectedSigma(PID::Deuteron),
134  resp.GetExpectedSigma(PID::Triton),
135  resp.GetExpectedSigma(PID::Helium3),
136  resp.GetExpectedSigma(PID::Alpha),
137  resp.GetNumberOfSigmas(PID::Electron),
138  resp.GetNumberOfSigmas(PID::Muon),
139  resp.GetNumberOfSigmas(PID::Pion),
140  resp.GetNumberOfSigmas(PID::Kaon),
141  resp.GetNumberOfSigmas(PID::Proton),
142  resp.GetNumberOfSigmas(PID::Deuteron),
143  resp.GetNumberOfSigmas(PID::Triton),
144  resp.GetNumberOfSigmas(PID::Helium3),
145  resp.GetNumberOfSigmas(PID::Alpha));
146  }
147  }
