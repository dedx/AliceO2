148 struct HalfChamberHeader {
149  union {
150  // 10987654321098765432109876543210
151  // uint32_t: 00000000000000000000000000000000
152  // cccccccccccccccc iiiiiiiiiii
153  // ffff| y|
154  // | | |------ 0-10 half chamber id
155  // | | ------------- 11 always 0x1
156  // | ----------------------------- 12-72 MCM Clock counter
157  // --------------------------------- 28-31 tracklet data format number
158  uint32_t word;
159 
160  struct {
161  uint32_t HCID : 11; // half chamber id 0:1079
162  uint32_t one : 1; //always 1
163  uint32_t MCLK : 16; // MCM clock counter 120MHz ... for simulation -- incrementing, and same number in all for each event.
164  uint32_t format : 4;
165  // 0 baseline PID 3 time slices, 7 bit each
166  // 1 DO NOT USE ! reserved for tracklet end marker disambiguation
167  // 14 Tracklet test-pattern mode
168  // 15 Reserved for testing
169  } __attribute__((__packed__));
170  };
171 };

175 // This constitutes the "4x32" bits of information from a single MCM, MCMRawDataHeader and 1-3 TrapRawTracklet.
176 struct MCMRawDataHeader {
177  //first word *
178  // 10987654321098765432109876543210
179  // uint32_t: 00000000000000000000000000000000
180  // 1zzzz pppppppp pppppppp1
181  // || yy| pppppppp | |--- 0 1 check bits
182  // || | | | ----------- 1-8 pid for tracklet 3 second part
183  // || | | -------------------- 9-16 pid for tracklet 2 second part
184  // || | ---------------------------- 17-24 pid for tracklet 1 second part
185  // || ------------------------------ 25-26 coly
186  // |---------------------------------- 27-30 padrow
187  // ----------------------------------- 31 1
188  //TODO need to check endianness, I have a vague memory the trap chip has different endianness to x86.
189  union {
190  uint32_t word;
191  struct {
192  uint32_t oneb : 1; //
193  uint32_t pid0 : 8; // 3 parts of pid for each tracklet
194  uint32_t pid1 : 8; // 3 parts of pid for each tracklet
195  uint32_t pid2 : 8; // 3 parts of pid for each tracklet
196  uint32_t col : 2; // 2 bits for position in pad direction.
197  uint32_t padrow : 4; // padrow,z coordinate for chip.
198  uint32_t onea : 1; //
199  } __attribute__((__packed__));
200  };

290  if (link < 15)
