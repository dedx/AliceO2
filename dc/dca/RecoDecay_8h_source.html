19 double invmass2prongs2(double px0, double py0, double pz0, double mass0,
20  double px1, double py1, double pz1, double mass1)
21 {
22 
23  double energy0_ = energy(px0, py0, pz0, mass0);
24  double energy1_ = energy(px1, py1, pz1, mass1);
25  double energytot = energy0_ + energy1_;
26 
27  double psum2 = (px0 + px1) * (px0 + px1) +
28  (py0 + py1) * (py0 + py1) +
29  (pz0 + pz1) * (pz0 + pz1);
30  return energytot * energytot - psum2;
31 };

33 double invmass3prongs2(double px0, double py0, double pz0, double mass0,
34  double px1, double py1, double pz1, double mass1,
35  double px2, double py2, double pz2, double mass2)
36 {
37  double energy0_ = energy(px0, py0, pz0, mass0);
38  double energy1_ = energy(px1, py1, pz1, mass1);
39  double energy2_ = energy(px2, py2, pz2, mass2);
40  double energytot = energy0_ + energy1_ + energy2_;
41 
42  double psum2 = (px0 + px1 + px2) * (px0 + px1 + px2) +
43  (py0 + py1 + py2) * (py0 + py1 + py2) +
44  (pz0 + pz1 + pz2) * (pz0 + pz1 + pz2);
45  return energytot * energytot - psum2;
46 };
