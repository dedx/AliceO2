607  : mAt{&at},
608  mGroupingElement{gt.begin()},
609  position{0}

611  auto indexColumnName = getLabelFromType();
612  arrow::compute::FunctionContext ctx;
616  auto splitter = [&](auto&& x) {
617  using xt = std::decay_t<decltype(x)>;
618  constexpr auto index = framework::has_type_at<std::decay_t<decltype(x)>>(associated_pack_t{});
619  if (hasIndexTo<std::decay_t<G>>(typename xt::persistent_columns_t{})) {
620  auto result = o2::framework::sliceByColumn(&ctx, indexColumnName,
621  static_cast<int32_t>(gt.size()),
622  x.asArrowTable(),
623  &groups[index],
624  &offsets[index]);
625  if (result.ok() == false) {
626  throw std::runtime_error("Cannot split collection");
627  }
628  if (groups[index].size() != gt.size()) {
629  throw std::runtime_error("Splitting collection resulted in different group number than there is rows in the grouping table.");
630  };
631  }

709  template <typename A1>
710  auto prepareArgument()
711  {
712  constexpr auto index = framework::has_type_at<A1>(associated_pack_t{});
713  if (hasIndexTo<G>(typename std::decay_t<A1>::persistent_columns_t{})) {
714  if constexpr (soa::is_soa_filtered_t<std::decay_t<A1>>::value) {
715  auto groupedElementsTable = arrow::util::get<std::shared_ptr<arrow::Table>>(((groups[index])[position]).value);
716 
717  // for each grouping element we need to slice the selection vector
718  auto start_iterator = std::lower_bound(starts[index], selections[index]->end(), (offsets[index])[position]);
719  auto stop_iterator = std::lower_bound(start_iterator, selections[index]->end(), (offsets[index])[position + 1]);
720  starts[index] = stop_iterator;
721  soa::SelectionVector slicedSelection{start_iterator, stop_iterator};
722  std::transform(slicedSelection.begin(), slicedSelection.end(), slicedSelection.begin(),
723  [&](int64_t idx) {
724  return idx - static_cast<int64_t>((offsets[index])[position]);
