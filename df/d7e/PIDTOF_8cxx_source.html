26 double Param::GetExpectedSigma(float mom, float time, float evtimereso, float mass) const
27 {
28  mom = abs(mom);
29  if (mom <= 0)
30  return -999;
31  double dpp = mPar[0] + mPar[1] * mom + mPar[2] * mass / mom; //mean relative pt resolution;
32  double sigma = dpp * time / (1. + mom * mom / (mass * mass));
33  return TMath::Sqrt(sigma * sigma + mPar[3] * mPar[3] / mom / mom + mSigma * mSigma + evtimereso * evtimereso);
34 }
