50  // bit masks for the above raw data;
51  static constexpr uint64_t formatmask = 0xf000000000000000;
52  static constexpr uint64_t hcidmask = 0x0ffe000000000000;
53  static constexpr uint64_t padrowmask = 0x0001e00000000000;
54  static constexpr uint64_t colmask = 0x0000180000000000;
55  static constexpr uint64_t posmask = 0x000007ff00000000;
56  static constexpr uint64_t slopemask = 0x00000000ff000000;
57  static constexpr uint64_t Q0mask = 0x0000000000ff0000;
58  static constexpr uint64_t Q1mask = 0x000000000000ff00;
59  static constexpr uint64_t Q2mask = 0x00000000000000ff;
60  static constexpr uint64_t PIDmask = 0x0000000000ffffff;
61  //bit shifts for the above raw data
62  static constexpr uint64_t formatbs = 60;
63  static constexpr uint64_t hcidbs = 49;
64  static constexpr uint64_t padrowbs = 45;
65  static constexpr uint64_t colbs = 43;
66  static constexpr uint64_t posbs = 32;
67  static constexpr uint64_t slopebs = 24;
68  static constexpr uint64_t PIDbs = 0;
69  static constexpr uint64_t Q0bs = 16;
70  static constexpr uint64_t Q1bs = 8;
71  static constexpr uint64_t Q2bs = 0;

76  Tracklet64(const Tracklet64& rhs);

78  uint64_t slope, uint64_t Q2, uint64_t Q1, uint64_t Q0)

87  //TODO convert to the actual number regarding compliments.
88  // ----- Getters for contents of tracklet word -----
89  uint64_t getHCID() const { return ((mtrackletWord & hcidmask) >> hcidbs); }; // no units 0..1077
90  uint64_t getPadRow() const { return ((mtrackletWord & padrowmask) >> padrowbs); }; // in units of
91  uint64_t getColumn() const { return ((mtrackletWord & colmask) >> colbs); }; // in units of
92  uint64_t getPosition() const { return ((mtrackletWord & posmask) >> posbs); }; // in units of 0.02 pads [10bits] .. -10.22 to 10.22
93  uint64_t getSlope() const { return ((mtrackletWord & slopemask) >> slopebs); }; // in units of -127 .. 127
94  uint64_t getPID() const { return ((mtrackletWord & PIDmask)); }; // in units of counts all 3 together
95  uint64_t getQ0() const { return ((mtrackletWord & Q0mask) >> Q0bs); }; // in units of counts all 3 together
96  uint64_t getQ1() const { return ((mtrackletWord & Q1mask) >> Q1bs); }; // in units of counts all 3 together
97  uint64_t getQ2() const { return ((mtrackletWord & Q2mask) >> Q2bs); }; // in units of counts all 3 together
98 
99  uint64_t buildTrackletWord(uint64_t format, uint64_t hcid, uint64_t padrow, uint64_t col, uint64_t position, uint64_t slope, uint64_t Q2, uint64_t Q1, uint64_t Q0)
100  {
101  mtrackletWord = ((format << formatbs) & formatmask) + ((hcid << hcidbs) & hcidmask) + ((padrow << padrowbs) & padrowmask) + ((col << colbs) & colmask) + ((position << posbs) & posmask) + ((slope << slopebs) & slopemask) + ((Q2 << Q2bs) & Q2mask) + ((Q1 << Q1bs) & Q1mask) + ((Q0 << Q0bs) & Q0mask);
102  return 0;
103  }
104  uint64_t setTrackletWord(uint64_t trackletword)
105  {
106  mtrackletWord = trackletword;
107  return 0;
108  }
109  // ----- Getters for tracklet information -----
110  // TODO figure out how to get MCM and ROB from hcid, col and padrow.
111  int getMCM() const
112  {
113  return (getColumn() % (FeeParam::mgkNcol / 2)) / FeeParam::mgkNcolMcm + FeeParam::mgkNmcmRobInCol * (getPadRow() % FeeParam::mgkNmcmRobInRow);
114  }
115  int getROB() const
116  {
117  int side = getColumn() / (FeeParam::mgkNcol / 2);
118  return (int)((int)getPadRow() / (int)FeeParam::mgkNmcmRobInRow) * 2 + side;
119  }
120 
121  // ----- Getters for offline corresponding values -----
122  int getDetector() const { return getHCID() / 2; }
123 
124  uint64_t getTrackletWord() const { return mtrackletWord; };

129  // TODO row and mcm to col and padrow mapping.

144  }
145  void setPID(uint64_t pid) { mtrackletWord |= ((((uint64_t)pid) << PIDbs) & PIDmask); } // set the entire pid area of the trackletword, all the 3 Q's
146  void setPosition(uint64_t position) { mtrackletWord |= ((((uint64_t)position) << posbs) & posmask); }
147  void setSlope(uint64_t slope) { mtrackletWord |= ((((uint64_t)slope) << slopebs) & slopemask); }
148

154 };
