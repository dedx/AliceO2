40 void CTFCoder::compress(CompressedDigits& cd, const gsl::span<const Digit>& digitVec, const gsl::span<const ChannelData>& channelVec)
41 {
42  // convert digits/channel to their compressed version
43  cd.clear();
44  if (!digitVec.size()) {
45  return;
46  }
47  const auto& dig0 = digitVec[0];
48  cd.header.nTriggers = digitVec.size();
49  cd.header.firstOrbit = dig0.getOrbit();
50  cd.header.firstBC = dig0.getBC();
51 
52  cd.trigger.resize(cd.header.nTriggers);
53  cd.bcInc.resize(cd.header.nTriggers);
54  cd.orbitInc.resize(cd.header.nTriggers);
55  cd.nChan.resize(cd.header.nTriggers);
56 
57  cd.idChan.resize(channelVec.size());
58  cd.qtc.resize(channelVec.size());
59  cd.cfdTime.resize(channelVec.size());
60  cd.qtcAmpl.resize(channelVec.size());
61 
62  uint16_t prevBC = cd.header.firstBC;
63  uint32_t prevOrbit = cd.header.firstOrbit;
64  uint32_t ccount = 0;
65  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
66  const auto& digit = digitVec[idig];
67  const auto chanels = digit.getBunchChannelData(channelVec); // we assume the channels are sorted
68 
69  // fill trigger info
70  cd.trigger[idig] = digit.getTriggers().triggersignals;
71  if (prevOrbit == digit.getOrbit()) {
72  cd.bcInc[idig] = digit.getBC() - prevBC;
73  cd.orbitInc[idig] = 0;
74  } else {
75  cd.bcInc[idig] = digit.getBC();
76  cd.orbitInc[idig] = digit.getOrbit() - prevOrbit;
77  }
78  prevBC = digit.getBC();
79  prevOrbit = digit.getOrbit();
80  // fill channels info
81  cd.nChan[idig] = chanels.size();
82  if (!cd.nChan[idig]) {
83  LOG(ERROR) << "Digits with no channels";
84  continue;
85  }
86  uint8_t prevChan = 0;
87  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
88  assert(prevChan <= chanels[ic].ChId);
89  cd.idChan[ccount] = chanels[ic].ChId - prevChan;
90  cd.qtc[ccount] = chanels[ic].ChainQTC;
91  cd.cfdTime[ccount] = chanels[ic].CFDTime;
92  cd.qtcAmpl[ccount] = chanels[ic].QTCAmpl;
93  prevChan = chanels[ic].ChId;
94  ccount++;
95  }
96  }
97 }
