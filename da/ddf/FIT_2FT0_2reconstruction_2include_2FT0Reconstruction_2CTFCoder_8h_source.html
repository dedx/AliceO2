61 template <typename VEC>
62 void CTFCoder::encode(VEC& buff, const gsl::span<const Digit>& digitVec, const gsl::span<const ChannelData>& channelVec)
63 {
64  using MD = o2::ctf::Metadata::OptStore;
65  // what to do which each field: see o2::ctd::Metadata explanation
66  constexpr MD optField[CTF::getNBlocks()] = {
67  MD::EENCODE, // BLC_trigger
68  MD::EENCODE, // BLC_bcInc
69  MD::EENCODE, // BLC_orbitInc
70  MD::EENCODE, // BLC_nChan
71  MD::EENCODE, // BLC_idChan
72  MD::EENCODE, // BLC_qtc
73  MD::EENCODE, // BLC_cfdTime
74  MD::EENCODE // BLC_qtcAmpl
75  };
76  CompressedDigits cd;
77  compress(cd, digitVec, channelVec);
78  auto ec = CTF::create(buff);
79  using ECB = CTF::base;
80 
81  ec->setHeader(cd.header);
82  ec->getANSHeader().majorVersion = 0;
83  ec->getANSHeader().minorVersion = 1;
84  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
85 #define ENCODE CTF::get(buff.data())->encode
86  // clang-format off
87  ENCODE(cd.trigger, CTF::BLC_trigger, o2::rans::ProbabilityBits16Bit, optField[CTF::BLC_trigger], &buff);
88  ENCODE(cd.bcInc, CTF::BLC_bcInc, o2::rans::ProbabilityBits16Bit, optField[CTF::BLC_bcInc], &buff);
89  ENCODE(cd.orbitInc, CTF::BLC_orbitInc, o2::rans::ProbabilityBits16Bit, optField[CTF::BLC_orbitInc], &buff);
90  ENCODE(cd.nChan, CTF::BLC_nChan, o2::rans::ProbabilityBits8Bit, optField[CTF::BLC_nChan], &buff);
91  ENCODE(cd.idChan , CTF::BLC_idChan, o2::rans::ProbabilityBits8Bit, optField[CTF::BLC_idChan], &buff);
92  ENCODE(cd.qtc , CTF::BLC_qtc, o2::rans::ProbabilityBits8Bit, optField[CTF::BLC_qtc], &buff);
93  ENCODE(cd.cfdTime, CTF::BLC_cfdTime, o2::rans::ProbabilityBits16Bit, optField[CTF::BLC_cfdTime], &buff);
94  ENCODE(cd.qtcAmpl, CTF::BLC_qtcAmpl, o2::rans::ProbabilityBits25Bit, optField[CTF::BLC_qtcAmpl], &buff);
95  // clang-format on
96 }

100 void CTFCoder::decode(const CTF::base& ec, VDIG& digitVec, VCHAN& channelVec)
101 {
102  CompressedDigits cd;
103  cd.header = ec.getHeader();
104  // clang-format off
105  ec.decode(cd.trigger, CTF::BLC_trigger);
106  ec.decode(cd.bcInc, CTF::BLC_bcInc);
107  ec.decode(cd.orbitInc, CTF::BLC_orbitInc);
108  ec.decode(cd.nChan, CTF::BLC_nChan);
109  ec.decode(cd.idChan , CTF::BLC_idChan);
110  ec.decode(cd.qtc , CTF::BLC_qtc);
111  ec.decode(cd.cfdTime, CTF::BLC_cfdTime);
112  ec.decode(cd.qtcAmpl, CTF::BLC_qtcAmpl);
113  // clang-format on
114  //
115  decompress(cd, digitVec, channelVec);
116 }

119 template <typename VDIG, typename VCHAN>
120 void CTFCoder::decompress(const CompressedDigits& cd, VDIG& digitVec, VCHAN& channelVec)
121 {
122  digitVec.clear();
123  channelVec.clear();
124  digitVec.reserve(cd.header.nTriggers);
125  channelVec.reserve(cd.idChan.size());
126 
127  uint32_t firstEntry = 0, clCount = 0, chipCount = 0;
128  o2::InteractionRecord ir(cd.header.firstBC, cd.header.firstOrbit);
129 
130  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
131  // restore ROFRecord
132  if (cd.orbitInc[idig]) { // non-0 increment => new orbit
133  ir.bc = cd.bcInc[idig]; // bcInc has absolute meaning
134  ir.orbit += cd.orbitInc[idig];
135  } else {
136  ir.bc += cd.bcInc[idig];
137  }
138  Triggers trig;
139  trig.triggersignals = cd.trigger[idig];
140 
141  firstEntry = channelVec.size();
142  uint8_t chID = 0;
143  int amplA = 0, amplC = 0, timeA = 0, timeC = 0;
144  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
145  auto icc = channelVec.size();
146  const auto& chan = channelVec.emplace_back((chID += cd.idChan[icc]), cd.cfdTime[icc], cd.qtcAmpl[icc], cd.qtc[icc]);
147  //
148  // rebuild digit
149  if (chan.ChId < 4 * uint8_t(Geometry::NCellsA)) { // A side
150  trig.nChanA++;
151  amplA += chan.QTCAmpl;
152  timeA += chan.CFDTime;
153  } else {
154  trig.nChanC++;
155  amplC += chan.QTCAmpl;
156  timeC += chan.CFDTime;
157  }
158  }
159  if (trig.nChanA) {
160  trig.timeA = timeA / trig.nChanA;
161  trig.amplA = amplA;
162  }
163  if (trig.nChanC) {
164  trig.timeC = timeC / trig.nChanC;
165  trig.amplC = amplC;
166  }
167  digitVec.emplace_back(firstEntry, cd.nChan[idig], ir, trig, idig);
168  }
169 }
