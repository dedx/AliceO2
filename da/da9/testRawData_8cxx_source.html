44 BOOST_AUTO_TEST_CASE(TRDRawDataHeaderInternals)
45 {
46  o2::trd::TrapRawTracklet tracklet;
47  o2::trd::HalfChamberHeader halfchamberheader;
48  o2::trd::HalfCRUHeader halfcruheader;
49  halfcruheader.word02[0] = 0x102;
50  BOOST_CHECK_EQUAL(halfcruheader.linksA[0].errorflags, 2);
51  BOOST_CHECK_EQUAL(halfcruheader.linksA[0].size, 1);
52  BOOST_CHECK_EQUAL(halfcruheader.linksA[0].errorflags, o2::trd::getlinkerrorflag(halfcruheader, 0));
53  BOOST_CHECK_EQUAL(halfcruheader.linksA[0].size, o2::trd::getlinkdatasize(halfcruheader, 0));
54  //ab is size
55  halfcruheader.word02[2] = 0xffffaa0000000000;
56  BOOST_CHECK_EQUAL(halfcruheader.linksA[7].size, 0xffff);
57  BOOST_CHECK_EQUAL(halfcruheader.linksA[7].errorflags, 0xaa);
58  BOOST_CHECK_EQUAL(halfcruheader.linksA[7].errorflags, o2::trd::getlinkerrorflag(halfcruheader, 7));
59  BOOST_CHECK_EQUAL(halfcruheader.linksA[7].size, o2::trd::getlinkdatasize(halfcruheader, 7));
60  halfcruheader.word3 = (0x222LL) << 40; //bc is at 0x000ffff000000000
61  BOOST_CHECK_EQUAL(halfcruheader.BunchCrossing, 0x222);
62  halfcruheader.word3 = 0x7700000000; //headerversion is at 0x00ff0000000
63  BOOST_CHECK_EQUAL(halfcruheader.HeaderVersion, 119);
64  halfcruheader.word57[0] = 0x345869faba;
65  halfcruheader.word57[2] = 0xaaade1277;
66  //linkb[0] zero is undefined.
67  BOOST_CHECK_EQUAL(halfcruheader.linksB[1].errorflags, 0x58);
68  BOOST_CHECK_EQUAL(halfcruheader.linksB[1].size, 0x34);
69  BOOST_CHECK_EQUAL(halfcruheader.linksB[1].errorflags, o2::trd::getlinkerrorflag(halfcruheader, 8));
70  BOOST_CHECK_EQUAL(halfcruheader.linksB[1].size, o2::trd::getlinkdatasize(halfcruheader, 8));
71  //ab is size
72  BOOST_CHECK_EQUAL(halfcruheader.linksB[5].size, 0x1277);
73  BOOST_CHECK_EQUAL(halfcruheader.linksB[6].size, 0xaaa);
74  BOOST_CHECK_EQUAL(halfcruheader.linksB[6].errorflags, 0xde);
75  BOOST_CHECK_EQUAL(halfcruheader.linksB[6].errorflags, o2::trd::getlinkerrorflag(halfcruheader, 13));
76  BOOST_CHECK_EQUAL(halfcruheader.linksB[6].size, o2::trd::getlinkdatasize(halfcruheader, 13));
77  //now test the boundary crossing of the int64_t
78  halfcruheader.word02[0] = (uint64_t)0x1000000000000000;
79  halfcruheader.word02[1] = (uint64_t)0xa00b000c000d0ebf;
80  BOOST_CHECK_EQUAL(halfcruheader.linksA[2].size, 0xbf10); // check a size that spans a 64bit word.
81  o2::trd::MCMRawDataHeader mcmrawdataheader;
82  mcmrawdataheader.word = 0x78000000;
83  BOOST_CHECK_EQUAL(mcmrawdataheader.padrow, 15);
84  mcmrawdataheader.word = 0x06000000;
85  BOOST_CHECK_EQUAL(mcmrawdataheader.col, 3);
86  mcmrawdataheader.word = 0x01fe0000;
87  BOOST_CHECK_EQUAL(mcmrawdataheader.pid2, 0xff); // 8 bits // gave up seperating pid, so its a flat 24 bit needs to be masked 0xfff 0xfff000 and 0xfff000000
88  mcmrawdataheader.word = 0x01fe;
89  BOOST_CHECK_EQUAL(mcmrawdataheader.pid0, 0xff); // 8 bits // gave up seperating pid, so its a flat 24 bit needs to be masked 0xfff 0xfff000 and 0xfff000000
90  //check tracklet
91  tracklet.word = 0xffc00000;
92  BOOST_CHECK_EQUAL((uint32_t)tracklet.pos, 0x3ff);
93 }
